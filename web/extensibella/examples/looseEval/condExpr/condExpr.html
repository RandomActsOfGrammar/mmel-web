Module looseEval:condExpr

Builds on looseEval:host

expr ::= ...
       | condExpr(expr, expr, expr) /*condExpr(C, T, F):  C ? T : F*/


------------------------------------------------------ [Proj-CondExpr]
|{expr}- condExpr(C, T, F) ~~>
         or(and(eq(C, true), or(eq(T, num(0)), true)),
            eq(F, num(0)))




vars C V1
vars T V2
vars F V3
V1 ++ V2 = V4
V4 ++ V3 = V
