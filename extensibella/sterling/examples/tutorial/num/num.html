tm ::= ...
     | num(int)
     | plus(tm, tm)

ty ::= ...
     | intTy

buildNumTerm I T
--------------------------------------- [Trans-Num]
|{tm}- num(I) ~~> abs("f", abs("x", T))


---------------------------------------------------------- [Trans-Plus]
|{tm}- plus(T1, T2) ~~>
       abs("m", abs("n", abs("f", abs("x",
          app(var("m"), app(var("f"),
              app(var("n"), app(var("f"), var("x")))))))))


---------------------------------------------------------- [Trans-IntTy]
|{ty}- intTy ~~> arrowTy(arrowTy(Ty, Ty), arrowTy(Ty, Ty))


/*Build the body of a Church encoding of a number,
  using names "f" and "x"*/
